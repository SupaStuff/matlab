ex 3.6

figure(2);
hold on;
x=-pi:pi/32:pi;
y=sin(x.^2).*cos(x);
dy=2.*x.*cos(x.^2).*cos(x) - sin(x.^2).*sin(x);
g=plot(x,y);
dg=plot(x,dy, 'r');
title(sprintf('Graph of y=sin(x^2)cos(x) and y''\nJean Peña'));
legend('y', 'y''');
set(g, 'LineStyle', '-');
set(dg, 'LineStyle', ':');
axis([-pi pi -6 6]);
grid on;
set(gca, 'XTick', -pi:pi/2:pi);
set(gca, 'XTickLabel', {'-pi'; '-pi/2'; '0'; 'pi/2'; 'pi'})
